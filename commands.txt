lunch single instance with security group and deploy the two different websites with same IPv4 address with different port number

git clone https://github.com
git clone https://githubtwo.com 

sudo mkdir -p /var/www/site1
sudo mkdir -p /var/www/site2

(go inside the folder) cp * /var/www/site1
(go inside the folder) cp * /var/www/site2

sudo nano /etc/nginx/sites-available/site1
sudo nano /etc/nginx/sites-available/site2

sudo ln -s /etc/nginx/sites-available/site1 /etc/nginx/sites-enabled/
sudo ln -s /etc/nginx/sites-available/site2 /etc/nginx/sites-enabled/

sudo nginx -t
sudo systemctl restart nginx

http://EC2_PUBLIC_IP:8080
http://EC2_PUBLIC_IP:9090
